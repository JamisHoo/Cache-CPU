        \hline
        Syntax: & slt \$d, \$s, \$t \\
        \hline
        Description: & If \$s is less than \$t, \$d is set to one. It gets zero otherwise. \\
        \hline
        Operation: & if \$s < \$t \$d = 1 else \$d = 0; \\
        \hline
        Encoding: & 0000 00ss ssst tttt dddd d000 0010 1010 \\
        \hline
    \end{tabularx}
    \end{table}
